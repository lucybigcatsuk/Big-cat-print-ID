  <h1>🐾 Big Cat Print Detector</h1>

  <div class="card">
    <input type="file" accept="image/*" capture="environment" id="upload">
    <img id="preview" />
    <button onclick="analyze()">Analyze Print</button>
    <div id="result"></div>
  </div>

  <script>
    let model;

    async function loadModel() {
      model = await tf.loadGraphModel(
        "https://tfhub.dev/google/tfjs-model/imagenet/mobilenet_v2_100_224/classification/5/default/1",
        { fromTFHub: true }
      );
      console.log("Model loaded");
    }

    loadModel();

    const upload = document.getElementById("upload");
    const preview = document.getElementById("preview");

    upload.addEventListener("change", () => {
      const file = upload.files[0];
      if (!file) return;
      preview.src = URL.createObjectURL(file);
    });

    async function analyze() {
      const result = document.getElementById("result");

      if (!preview.src) {
        result.innerText = "Please upload a photo first.";
        return;
      }

      result.innerText = "Analyzing...";

      const img = new Image();
      img.src = preview.src;

      img.onload = async () => {
        const tensor = tf.browser.fromPixels(img)
          .resizeNearestNeighbor([224, 224])
          .toFloat()
          .expandDims();

        const predictions = await model.predict(tensor).data();

        const confidence = Math.max(...predictions);

        if (confidence > 0.5) {
          result.innerText = "🐾 Possibly a big cat print";
        } else {
          result.innerText = "❌ Likely not a big cat print";
        }
      };
    }
  </script>
